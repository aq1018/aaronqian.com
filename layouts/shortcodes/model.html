{{- /* Embed a glTF/GLB model via Google's model-viewer web component. Args: src (required, relative to page bundle), alt, aspect (default 16/9), tilt (degrees down from horizontal; negative looks from above). */ -}} {{- $src := .Get "src" -}} {{- $alt := .Get "alt" | default "3D model" -}} {{- $aspect := .Get "aspect" | default "16/9" -}} {{- $tilt := .Get "tilt" | default "0" -}} {{- $resource := .Page.Resources.GetMatch $src -}} {{- $url := cond (ne $resource nil) ($resource.RelPermalink) $src -}} {{- /* model-viewer camera-orbit: theta phi radius. Default phi 75deg, tilt subtracts. */ -}} {{- $phi := sub 75 (int $tilt) -}} {{- /* Load the model-viewer web component once per page on first use. */ -}} {{- if not (.Page.Store.Get "modelViewerLoaded") -}} {{- .Page.Store.Set "modelViewerLoaded" true -}} {{- end -}}
