{{- /* Prev / next navigation within the current section. Accepts either a Page directly, or a dict {page, position} where position is "top" or "bottom" (default "bottom"). */ -}} {{- $page := . -}} {{- $position := "bottom" -}} {{- if reflect.IsMap . -}} {{- $page = .page -}} {{- $position = .position | default "bottom" -}} {{- end -}} {{- $prev := $page.PrevInSection -}} {{- $next := $page.NextInSection -}} {{- if or $prev $next -}}
{{- if $prev -}} ← Previous {{ $prev.Title }} {{- else -}} {{- end -}} {{- if $next -}} Next → {{ $next.Title }} {{- else -}} {{- end -}}
{{- end -}}
